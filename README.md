    padding: 4% 26%;
    color: white;
    background-color: #382AE1;
    font-family: $font-stack;
    font-weight: 700;
    border: none;
    filter: drop-shadow(0px 15px 10px #BDB8F4);
      &:active {
        background-color: #698DFF;
      }
}
